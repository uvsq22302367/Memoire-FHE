\section{Introduction au FHE}
% ==========================================

\begin{frame}{Le paradoxe de la confidentialité}
    \begin{block}{Le problème classique}
        Traditionnellement, la donnée est soit \textbf{sécurisée} (chiffrée) soit \textbf{traitable} (en clair). Calculer sur des données sensibles nécessite un déchiffrement en RAM, créant une faille.
    \end{block}
    
    \vspace{0.5cm}
    
    \begin{itemize}
        \item \textbf{La solution :} Le \textit{Fully Homomorphic Encryption} (FHE).
        \item \textbf{Objectif :} Évaluer des fonctions mathématiques arbitraires directement sur les cryptogrammes.
        \item \textbf{La rupture (2009) :} Craig Gentry prouve l'existence d'un tel schéma grâce au concept de \textit{bootstrapping}.
    \end{itemize}
\end{frame}

\begin{frame}{Modèle de calcul : Les circuits booléens}
    Les ordinateurs n'évaluent pas des fonctions mathématiques complexes, mais des réseaux de \textbf{portes logiques}.
    
    \vspace{0.5cm}
    
    \begin{block}{Universalité sur $\mathbb{F}_2$}
        Toute fonction calculable peut être traduite avec l'ensemble \(\{ \text{AND}, \text{XOR} \}\).
        \begin{itemize}
            \item \textbf{XOR} $\iff$ Addition modulo 2
            \item \textbf{AND} $\iff$ Multiplication modulo 2
        \end{itemize}
    \end{block}

    \vspace{0.2cm}
    \textit{Conséquence :} Si un cryptosystème permet d'additionner et de multiplier des chiffrés, il peut évaluer \textbf{n'importe quel programme}.
\end{frame}

\section{DGHV : Chiffrement sur les entiers}
% ==========================================

\begin{frame}{Le schéma DGHV (2010)}
    \textbf{Idée principale :} Remplacer les mathématiques complexes des réseaux (Gentry) par l'arithmétique des entiers.
    
    \begin{block}{Le Problème AGCD}
        Retrouver la clé secrète $p$ à partir de "presque-multiples" :
        \[ x_i = p \cdot q_i + r_i \]
        Le \textbf{bruit} ($r_i$) est indispensable : sans lui, un simple PGCD brise le système.
    \end{block}

    \begin{itemize}
        \item \textbf{Chiffré :} $c = q \cdot p + 2r + b$
        \item \textbf{Déchiffrement :} $b = (c \bmod p) \bmod 2$
    \end{itemize}
\end{frame}

\begin{frame}{Le problème du bruit et le Bootstrapping}
    \begin{alertblock}{L'explosion du bruit}
        Chaque multiplication de chiffrés double la taille du bruit. S'il dépasse $p/2$, le déchiffrement échoue. Le schéma est limité (\textit{Somewhat Homomorphic}).
    \end{alertblock}

    \vspace{0.3cm}
    
    \textbf{La solution : Le Bootstrapping}
    \begin{itemize}
        \item Évaluer le circuit de déchiffrement \textit{homomorphiquement} pour "nettoyer" le chiffré.
        \item \textbf{Problème :} La division $(c \bmod p)$ est trop lourde.
        \item \textbf{Le Squashing :} Remplacer la division par une somme de sous-ensembles (vecteur creux) avec $15$ additions environ.
    \end{itemize}
\end{frame}

\section{CKKS : Calculs approchés}
% ==========================================

\begin{frame}{Le schéma CKKS (2017)}
    \begin{block}{Un changement de paradigme}
        Contrairement à DGHV, CKKS accepte que le résultat ne soit pas exact. C'est un schéma \textbf{approximatif} dédié aux nombres réels/complexes.
    \end{block}
    
    \vspace{0.5cm}
    
    \textbf{Domaines d'application idéaux :}
    \begin{itemize}
        \item Machine Learning privé.
        \item Traitement de signal, données statistiques.
        \item Tout domaine où une erreur d'arrondi est tolérée (virgule flottante).
    \end{itemize}
\end{frame}

\begin{frame}{Encodage et gestion de l'échelle (Rescaling)}
    \textbf{L'anneau de travail :} $R_q = \mathbb{Z}_q[X]/(X^N+1)$
    
    \vspace{0.3cm}
    
    \begin{itemize}
        \item Les données (vecteurs complexes) sont encodées dans un polynôme via une erreur d'arrondi acceptée.
        \item On utilise un \textbf{facteur d'échelle} $\Delta$.
    \end{itemize}

    \begin{alertblock}{Le Rescaling}
        Après une multiplication homomorphe, l'échelle devient $\Delta^2$. Il faut diviser le chiffré par $\Delta$ pour éviter la saturation (équivalent à la troncature en virgule flottante).
    \end{alertblock}
\end{frame}

\section{Bilan et Ouverture}
% ==========================================

\begin{frame}{Conclusion \& Ouverture}
    \begin{itemize}
        \item \textbf{DGHV :} Pédagogique, élégant mathématiquement, mais complexe à rendre efficace (le bruit détruit l'information).
        \item \textbf{CKKS :} Pragmatique, sacrifie l'exactitude absolue pour une efficacité redoutable sur les données du monde réel.
    \end{itemize}

    \vspace{0.5cm}
    
    \begin{block}{Ouverture : Le schéma TFHE (2016)}
        Calcul sur le Tore réel ($\mathbb{R}/\mathbb{Z}$). Bootstrapping radicalement optimisé : permet d'évaluer des portes logiques en \textbf{millisecondes}. Ouvrant la voie aux microprocesseurs chiffrés.
    \end{block}
